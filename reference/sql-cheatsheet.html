SQL Cheatsheet

Ang mga pinakagamit na SQL commands — laging reference!

Basic Queries

-- Pumili ng lahat ng columns
SELECT * FROM users;

-- Pumili ng specific columns
SELECT name, email FROM users;

-- I-filter gamit WHERE
SELECT * FROM users WHERE age >= 18;

-- I-sort gamit ORDER BY
SELECT * FROM products ORDER BY price DESC;

-- Limitahan ang results
SELECT * FROM users LIMIT 10;

Insert, Update, Delete

-- Mag-insert ng bagong record
INSERT INTO users (name, email, age)
VALUES ('Juan', 'juan@email.com', 25);

-- Mag-update ng existing record
UPDATE users SET age = 26 WHERE name = 'Juan';

-- Mag-delete ng record
DELETE FROM users WHERE id = 5;

Filtering

-- LIKE — pattern matching
SELECT * FROM users WHERE name LIKE '%Juan%';

-- BETWEEN — range
SELECT * FROM products WHERE price BETWEEN 10 AND 100;

-- IN — multiple values
SELECT * FROM users WHERE age IN (18, 21, 25);

Aggregate Functions

-- COUNT — bilang ng records
SELECT COUNT(*) FROM users;

-- SUM — total ng values
SELECT SUM(price) FROM orders;

-- AVG — average
SELECT AVG(price) FROM products;

-- GROUP BY — group at i-aggregate
SELECT department, COUNT(*) FROM employees GROUP BY department;

-- HAVING — filter pagkatapos ng GROUP BY
SELECT department, AVG(salary) FROM employees
GROUP BY department HAVING AVG(salary) > 50000;

Joins

-- INNER JOIN — records na may match sa both tables
SELECT users.name, orders.total
FROM users
INNER JOIN orders ON users.id = orders.user_id;

-- LEFT JOIN — lahat ng nasa left table, kahit walang match
SELECT users.name, orders.total
FROM users
LEFT JOIN orders ON users.id = orders.user_id;

Create Table

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(255) UNIQUE,
    age INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
