Loops

# For loop — paulit-ulit sa collection
for item in list:
    print(item)

# For loop gamit range
for i in range(5):  # 0, 1, 2, 3, 4
    print(i)

# While loop — hanggang sa condition ay True
while count < 10:
    print(count)
    count += 1

Functions

def function_name(param1, param2):
    """Docstring — explanation ng function."""
    result = param1 + param2
    return result

# Tawagin ang function
output = function_name(3, 5)

File Handling

# Magbasa ng file
with open("file.txt", "r") as f:
    content = f.read()

# Magsulat ng file
with open("file.txt", "w") as f:
    f.write("Hello, World!")

# Magbasa ng CSV
import csv
with open("data.csv", "r") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)
